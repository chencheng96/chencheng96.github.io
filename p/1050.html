controller，提供前后端连接的接口 


package vip.chencheng.controller;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.ResponseBody;

import vip.chencheng.pojo.UrlList;
import vip.chencheng.service.UrlService;


@Controller
@RequestMapping(value = "/buy")
public class UrlController {
	
	@Autowired 
	private UrlService urlService;
	
	//提交新设备申请
	@RequestMapping(value = "/find", method = RequestMethod.GET)
    public @ResponseBody List<UrlList> find(String status) {
    	System.out.println("in controller");
    	
    	urlService.find(status);
    	return null;
    }
}


service，接口 


package vip.chencheng.service;

import java.util.List;

import vip.chencheng.pojo.UrlList;

public interface UrlService {

	public List<UrlList> find(String status);
}




package vip.chencheng.service.impl;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import vip.chencheng.dao.UrlDao;
import vip.chencheng.dao.impl.UrlDaoImpl;
import vip.chencheng.pojo.UrlList;
import vip.chencheng.service.UrlService;

@Service
public class UrlServiceImpl implements UrlService {
	
	@Autowired
	private UrlDao urlDao;
	
	//private UrlDaoImpl urlDao = new UrlDaoImpl();
	
	public List<UrlList> find(String status) {
		System.out.println("service imple "+status);
		//建立持久化对象
        UrlList url=new UrlList();
        url.setName("小红");
        url.setKind(status);

		urlDao.findAll(url);
		return null;
	}
}
