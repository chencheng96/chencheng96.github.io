表单验证

<script type="text/javascript">
function check(){
if(document.form1.xuehao.value==""){window.alert("请输入学号");document.form1.xuehao.focus();return false;}     //光标在此处停留，等待用户输入
if(document.form1.name.value==""){window.alert("请输入姓名");document.form1.name.focus();return false;}     //光标在此处停留，等待用户输入
if(document.form1.pass.value==""){window.alert("请输入密码");document.form1.pass.focus();return false;}     //光标在此处停留，等待用户输入
if(document.form1.pass.value.length<=6){window.alert("密码长度要大于6位");document.form1.pass.focus();return false;}     //光标在此处停留，等待用户输入
if(document.form1.pass.value!=document.form1.pass1.value){window.alert("两次密码不相同");document.form1.pass1.focus();return false;}     //光标在此处停留，等待用户输入
}
</script>
<form name="form1" method="post" action="" onsubmit="return check()">
<input type="text" name="xuehao"/>
<input type="text" name="name"/>
<input type="text" name="pass"/>
<input type="text" name="pass1"/>
<input type="submit" name="submit" value="提交">
</form>

表单的提交方式

通常提交表单的内容到一个动态网页，动态网页接受表单内容后，将内容保存到服务器上。另一种处理情况是提交后，将内容作为邮件的内容发送到一个指定的邮件。

<script type="text/javascript">
function check(){
if(document.form1.submittype.value=="server"){document.form1.action="http://www.iyaka.cn/main/extend/online_test.asp";}     //如果选择提交到服务器，将表单内容提交到动态网页处理。
else{document.encoding="text/plain";document.form1.action="mailto:123@163.com";}
}
</script>
<form name="form1" method="post" action="" onsubmit="return check()">
姓名：<input type="text" name="name"/><br/>
提交方式：<select name="submittype">
<option value="server">提交到服务器</option>
<option value="email">提交到邮箱</option>
</select></br>
<input type="submit" name="submit" value="提交">
<input type="reset" name="submit2" value="重置">
</form>

重置表单的提示

用户单击重置按钮，浏览器会把表单中的所有元素的值设置为初始状态。为防止用户误点击，可以弹出一个确认框。

通常提交表单的内容到一个动态网页，动态网页接受表单内容后，将内容保存到服务器上。另一种处理情况是提交后，将内容作为邮件的内容发送到一个指定的邮件。

<script type="text/javascript">
function check(){
if(window.confirm("您真的要重置表单吗？")){return true;}     //如果用户选择了是，继续操作，重置表单
else{return false;}
}
</script>
<form name="form1" method="post" action="" onreset="return check()">        //onsubmit决定是否提交，onrest绝对是否重置。
姓名：<input type="text" name="name"/><br/>
<input type="submit" name="submit" value="提交">
<input type="reset" name="submit2" value="重置">
</form>
