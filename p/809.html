实现跳转到新页面所需要的相关文件
1. 在AndroidManifest.xml中需要声明要跳转页面的类MemberActivity。 
2. 在WangYiActivity.java中需要设置当前页面的布局文件，按钮的初始化，通过id获取按钮对象，判断点击事件的按钮以判断将要执行的操作。最后需要设置跳转页面的Activity类。
3. 在将要跳转到的Activity中也要声明页面的布局文件的路径。

button控件（在布局页面画出来的控件）的初始化及事件的响应
   需要初始化button然后通过布局文件设置的控件id来获取对象，然后在onClick事件中int id = v.getId(); 获取点击事件的id,然后处理对应的事件。

下面文件为WangYiActivity.java
package cn.iyaka.mytext.wangyi; 
//所在包
import android.app.Activity;
import android.content.Intent;
import android.os.Bundle;
import android.support.annotation.Nullable;
import android.view.View;
import android.widget.Button;

//一些包的引入

import cn.iyaka.mytext.R;
/**
* Created by 19829 on 2018/1/6.
*/
public class WangYiActivity extends Activity implements View.OnClickListener {
Button buttonmember;                      //在类中定义一个button
@Override
protected void onCreate(@Nullable Bundle savedInstanceState) {
super.onCreate(savedInstanceState);                 //输入onCreate找到对于内容回车，自动导入所需要的包
setContentView(R.layout.wangyi_main);           //本页面的布局文件

buttonmember=(Button) findViewById(R.id.button_to_member);        //在onCreate方法中 获取button的对象，通过button的id
buttonmember.setOnClickListener(this);          //设置button的监听事件
}

@Override                                                 //方法的覆盖
public void onClick(View v) {

int id = v.getId();                            //获取onClick对象的id
switch(id) {
    case R.id.button_to_member:               //多个按钮时用switch...case判断点击的按钮
        Intent intent=new Intent();
        intent.setClass(this, MemberActivity.class);               //设置要跳转到页面的Activity
        startActivity(intent);                          //打开新的页面
        break;
    default:
        break;
}
}
}
