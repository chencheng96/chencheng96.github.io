1.实现点击按钮，复制文本框中的的内容（在谷歌和360浏览器上正常运行，在IE浏览器上显示Internet Explorer 已对此页面进行了修改，以帮助阻止跨站脚本）

在下一篇文章中将重点介绍一下document.execCommand("Copy"); // 执行浏览器复制命令实现较多的复制到剪切板功能的方法及实例

<script type="text/javascript">
function copyUrl2()
{
var Url2=document.getElementById("biao1");
Url2.select(); // 选择对象
document.execCommand("Copy"); // 执行浏览器复制命令
alert("已复制好，可贴粘。");
}
</script>
<textarea cols="20" rows="10" id="biao1">用户定义的代码区域</textarea>
<input type="button" onClick="copyUrl2()" value="点击复制代码" />

5.复制 span 标记中的内容（此种方法在360及谷歌浏览器上测试无法正常复制，在IE浏览器上正常）

<script type="text/javascript">
function copyText(obj)  
{
var rng = document.body.createTextRange();
rng.moveToElementText(obj);
rng.scrollIntoView();
rng.select();
rng.execCommand("Copy");
rng.collapse(false);
alert("复制成功!");
}
</script>
<span id="span">
    <span style="color: rgb(229, 51, 51);">2132131321</span>
</span>
<input type="button" onClick="copyText(span)" value="点击复制代码" />
