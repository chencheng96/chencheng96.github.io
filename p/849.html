运算符重载就是在函数体中重新定义运算符，对该运算符赋予多重含义。

运算符重载的两种形式：重载为类的成员函数，重载为类的友元函数。

重载为类的成员函数

#include <iostream>
using namespace std;
class CharSum
{
	char x[20];
public:
	CharSum(){}
	CharSum(char s[]){strcpy(x, s);}
	CharSum operator + (CharSum s)
	{
		int i=0, j=0;
		while(x[i] != '\0')i++;
		while((x[i]=s.x[j])!='\0')
		{
			j++;
			i++;	
		}
		return x;
	}
	void disp()
	{
		cout << x << endl;
	}
};
void main()
{
	CharSum str1("my "),str2("name");
	CharSum str3=str1+str2 ;
	str3.disp();
}
