数组

int a[10];

数组中元素的个数必须是常量（5）和符号常量（3+5）。不能包含变量（n）。在调用函数中定义数组，其长度可以是变量或非常量表达式（形参从实参得到确定的值）。

一维数组的初始化

可以在定义时对元素赋值： 

int a[3]={1,2,2};
int b[5]={1,2};//其他元素值为0
int c[]={1,2};//相当于定义了c[2]

多维数组

int a[3][4]={{1},{1},{1}};    //按行进行赋值的，定义数组的第一列全为1，其他3列为0。

字符数组

定义同一维数组（仅是不同的数据类型）

字符串

char c[]={"I am happy"};    //可以省略花括号，该数组长度为11，字符串常量的最后由系统自动加上'\0'。初始化赋值，或逐个对元素赋值，不能char s[11];s={"I am happy"};

字符数组可以逐个字符输出，也可以将整个字符串输出‘

  • printf("%s\n",c);    //    输出数组c中的字符串,输出字符串时遇到'\0'就停止。
  • scanf("%s",c);    //    输入一个字符串，输入项如果是字符数组名，不再加地址符&。
  • puts(字符数组);
  • gets(字符数组);
  • strcat(字符数组1, 字符数组2);    //字符串连接函数
  • strcpy(字符数组1, 字符数组2);    //将字符串2复制到字符串1中去
  • strncpy(str1,str2,n);    //将str2前面n个字符复制到str1中，n不应该多余str1中原有的字符个数
  • strcmp(str1,str2);    //比较字符串1和字符串2。相等返回0，str1>str2返回一个正整数
  • strlen(str);    //测试字符串长度，不含'\0'
  • strlwr(str);    //大写字母换成小写字母
  • strupr(str);    //小写转换为大写

字符串处理函数应包含头文件 <string.h>

指针 

#include<stdio.h>
void main()
{
	int a=100,b=10;
	int *p1,*p2;	//定义两个整形指针
	p1=&a;	//a的地址赋值给指针变量
	p2=&b;
	printf("a=%d,b=%d\n",a,*p2);	//*p2获得指针变量所指地址存放的值
}

指针变量做函数参数 

#include<stdio.h>
void main()
{
	void swap(int *pt1,int *pt2);
	int a=100,b=10;
	int *p1,*p2;	//定义两个整形指针
	p1=&a;	//a的地址赋值给指针变量
	p2=&b;
	swap(p1,p2);
	printf("a=%d,b=%d\n",a,*p2);	//*p2获得指针变量所指地址存放的值
}
void swap(int *pt1,int *pt2)
{
	int temp;
	temp=*pt1;
	*pt1=*pt2;
	*pt2=temp;
}
实参和形参之间数据的传递是值传递。调用函数传递指针也不能改变实参指针变量的值，但是可以改变实参指针所指变量的值。

指针引用数组 

#include<stdio.h>
void main()
{
	int a[10]={1,2,3,4,5,6};
	int *p;
	p=&a[0];	//效果同p=a,因为数组名就代表数组第一个元素的地址
	p++;	//数组指针可以进行++，--，+，-运算。p++指数组下一个元素的地址
	printf("%d\n",*p);
}

会输出数组的第二个元素 2 。

数组名做函数的参数传递的是地址，所以对形参的更改会影响实参。

指针引用多维数组 

#include<stdio.h>
void main()
{
	int a[10][2]={1,2,3,4,5,6};
	printf("%d\n",a[1]+2);	//第2行第3列的地址
	printf("%d\n",*(a+1)+2);	//第2行第3列的地址
}

对于多维数组a[i][j]等价于 *(a[i]+j) 或 *( *(a+i) +j)

a[i]与*(a+i) 以及a+1都表示第i行的起始地址

指针引用字符串 

#include<stdio.h>
void main()
{
	char *str1="I LOVE ";
	char str2[]="china";
	char *temp;
	printf("%s%s\n",str1,str2);
	temp=str2;	//将str2字符串地址赋值给temp
	printf("%s\n%c\n",temp,temp[3]);	//通过temp输出str2中的内容
	printf("%d\n",temp);	//输出str2首元素的地址
	str2[2]='s';
	//str1[3]='m';	//字符串常量的内容不能改变，该句无法执行
	printf("%s\n",temp);
}

函数指针

可以定义一个指向函数的指针变量，存放某一函数的起始地址。 

#include<stdio.h>
void main()
{
	int max(int,int);
	int (*p)(int,int);	//定义一个函数指针
	int a=7,b=8,c;
	p=max;	//指针指向函数
	c=(*p)(a,b);		//通过指针调用函数
	printf("%d\n",c);
}
int max(int x,int y)
{
	return x>y?x:y;
}

用函数指针做函数参数 

#include<stdio.h>
void main()
{
	int max(int,int);
	int min(int,int);
	int fun(int i, int j, int (*x2)(int,int));
	//int (*p)(int,int);	//定义一个函数指针
	int c,d;
	//p=max;	//指针指向函数
	c=fun(9,2,max);		//通过指针传递函数，也可以直接用p
	d=fun(9,2,min);
	printf("%d  %d\n",c,d);
}
int fun(int i, int j, int (*x2)(int,int)) 
{
	return (* x2)(i,j);
}
int max(int x,int y)
{
	return x>y?x:y;
}
int min(int x,int y)
{
	return x<y?x:y;
}

指针数组

数组元素均为指针类型的数据，即数组中每个元素都是一个地址。

int *p[4];    //[]优先级比*高

int ( *p )[4];    //指向一维数组的指针变量 

#include<stdio.h>
void main()
{
	int i;
	char * name[]={"firsrt","second","third"};
	char **p;
	for(i=0;i<3;i++)
	{
		p=name+i;
		printf("%s	",*p);
	}
}

动态内存分配

需要通过 stdlib.h 声明

  • malloc函数：void *malloc(unsigned int size);    //生成一个长度为size的连续空间。返回第一个字节的指针，失败返回null
  • calloc函数：void *calloc(unsigned n, unsigned size);    //分配n个长度为size的连续空间。成功返回起始位置指针，失败返回null
  • free函数：void free(void *p);    //释放指针变量p指向的动态空间
  • realloc函数：void *realloc(void *p, unsigned int size);    //改变由malloc和calloc生成空间的大小。将p指向空间的大小改为size，p不变，失败返回null

void指针类型 

#include<stdio.h>
void main()
{
	int a=3;
	void *p;
	int *m,*v;
	m=&a;
	p=&a;
	printf("m指向的：%d\n",*m);
	//printf("p指向的：%d\n",*p);	//p并不指向a，只是把a的地址赋值给p
	printf("m的内容：%d，p的内容：%d\n",m,p);
	v=p;
	printf("v指向的：%d\n",*v);	//可以把void类型的指针存放的地址赋值给合适类型的指针
}
