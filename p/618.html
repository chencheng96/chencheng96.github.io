装载文档（onload）和卸载文档（onunload）

一般作用在body标签中。

当打开浏览器窗口时就会激发装载文件

激发卸载事件的3种情况：

1.当浏览器从一个内容跳转到另一个内容时

2.浏览器窗口关闭时

3.刷新浏览器窗口时

<script type="text/javascript">
function on(){window.alert("欢迎访问网站");}
function un(){window.alert("谢谢！欢迎下次再来");}
</script>
<body onload="on()" onunload="un()"/>

得到焦点（focus）和失去焦点（blur）

焦点指浏览器窗口为当前活动窗口。当浏览器窗口得到焦点时激发onfocus事件，当浏览器窗口失去焦点时激发onblur事件。

<script type="text/javascript">
function on(){
if(document.myform.mytext.value=="输入内容")document.myform.mytext.value="";
}
function un(){
if(document.myform.mytext.value=="")document.myform.mytext.value="输入内容";
}
</script>
<form name="myform">
<input type="text" name="mytext" value="输入内容" onfocus="on()" onblur="un()">
</form>

调整窗口大小

当浏览器窗口被调整大小时，激发onresize事件。

<script type="text/javascript">
function win(){
if(window.confirm("是否把窗口调整到1024*768大小？"))window.resizeTo(1024,768);
}
</script>
<body onresize="win()">

错误事件

window对象的错误事件由浏览器产生。如果发生JavaScript代码错误，会在浏览器左下角出现提示，双击错误区域，可以看到详细错误信息。

error事件可以由下列3个参数给出错误处理函数

  • 详细的错误信息
  • 参数错误的的网页地址
  • 产生错误的行数

<script type="text/javascript">
window.onerror=check;  //直接在window对象中调用错误事件
function check(e1,e2,e3){
var str="发生错误信息\n";
str+="错误信息："+e1+"\n";
str+="发生错误的文件是："+e2+"\n";
str+="错误代码的行数："+e3+"\n";
window.alert(str);
return true;
}
var err=a+b;      //a+b没有定义
</script>
