meta标签 分为http-equiv属性和name属性

  • <meta http-equiv="Content-Type" content="text/html; charset=gb2312" /> //指定网页字符集
  • <mata name="keywords" content="网页设计"/> //指定页面关键字
  • <mata name="description" content="网页设计"/> //页面说明
  • <meta http-equiv="refresh" content="2" /> //2秒内刷新当前页
  • <meta http-equiv="refresh" content="2;URL=http://www.iyaka.cn" /> //2秒内跳转到指定网址
  • <mata name="robots" content="all"/> //robots，告诉搜索机器人哪些页面需要索引。content值：all,none,index,noindex,follow,nofollow.默认all
  • <mata name="author" content="chen"/> //作者
  • <mata name="copyright" content="© 2017-2018网页"/> //版权
  • <meta http-equiv="Page-Enter" content="revealtrans(duration=3,transition=23)" > //进入网页的动态效果
  • <meta http-equiv="Page-Exit" content="revealtrans(duration=3,transition=23)" > //退出网页的动态效果
