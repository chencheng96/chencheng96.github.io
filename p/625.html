设置链接颜色

默认情况，链接没有访问为蓝色，已访问过的链接和正在访问的链接颜色为暗红色，document对象的linkColor,vlinkColor和alinkColor属性可以分别设置这三个属性。

设置网页背景颜色和文字颜色

document对象的bgColor属性可以设置网页背景颜色，fgColor属性可以设置网页默认文字颜色。

<script type="text/javascript">
function set(){
document.alinkColor=document.f1.b1.value;
document.linkColor=document.f1.b2.value;
document.vlinkColor=document.f1.b3.value;
document.bgColor=document.f1.b4.value;
document.fgColor=document.f1.b5.value;
}
</script>

文档信息

使用lastModified属性查看文档最后修改时间，通过title和URL属性可以查看网页文档的标题和地址

<script type="text/javascript">
document.write("网页文档的最后修改时间："+document.lastModified+"</br>");
document.write("网页文档的标题："+document.title+"</br>");
document.write("网页文档的地址："+document.URL+"</br>");
</script>

在标题栏和状态栏显示滚动信息

可以在网页标题栏和状态栏显示文本信息，文本信息保存在数组中，循环显示数组中的元素。

<script type="text/javascript">
var i=0;
var sarr=["html","css","javascript"];
function showstatus(){
if(i==sarr.length)i=0;   //如果已经显示完毕，让数组下标为0，重新开始
document.title=sarr[i];  //循环显示数组的信息在状态栏上
window.status=sarr[i]; //循环显示数组信息在状态栏
i++;            //计数器增加1，显示下一条数组的元素
document.write("a</br>");
}
window.setInterval("showstatus()",2000); //设置循环显示的时间间隔为2秒
</script>

防止盗链

所以document对象的URL属性和referrer属性防止盗链行为。

<script type="text/javascript">
function plink()
{var cURL=document.URL;   //当前文档的链接
var fURL=document.referrer;   //上一个文档的链接
if(fURL!="")       //如果上一个文档为空，说明是直接打开的，否则可能是盗链
{var cURLs=cURL.split("/");     //当前文档各部分URL存放在cURLs数组中
var fURLs=fURL.split("/");      //上一个文档的URL各部分存放在fURLs数组中
if(cURLs[2]==fURLs[2])document.write("不是盗链"); //比较两数组的第三个元素，相同则不是盗链
else{
document.write("这是盗链");
history.location="http://"+cURLs[2];  //跳转到网站首面
}
}
else document.write("您是直接打开的文件");
}
plink();
</script>

在网页中输出内容

document.write(v1,v2…);

document.writeln(v1,v2…);

可以有一个或者多个参数，也可以没有参数。参数类型为字符串，如果不是字符串则自动转换为字符串。后者输出后会自动换行。
